#define PWM_MAX_VALUE   0xFF

Maximum number of sample levels. Larger numbers will result in larger PWM period

#define PWM_MIN_VALUE   0x0003

Minimum number of sample levels. Values lower than this will result in no PWM output

#define PWM_OUTPUT_PIN_NUMBER_1   BSP_LED_0
